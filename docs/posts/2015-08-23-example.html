/// simple binomial sampler that writes the results in a given buffer
fn binomial_sample(buffer: &mut Vec<f64>, p: f64, rng: &mut SmallRng) {
    for i in 0..buffer.len() {
        let x: f64 = rng.gen();
        if x > p {
            buffer[i] = 0.0;
        } else {
            buffer[i] = 1.0;
        }
    }
}
